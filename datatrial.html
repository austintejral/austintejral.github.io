starrecruits <- recruits |>
  group_by(Team) |>
  mutate(
    TFStotalfives = sum(TFSFiveStars),
    TFStotalfours = sum(TFSFourStars),
    TFStotalthrees = sum(TFSThreeStars),
    TFSthreestartotal = TFStotalthrees * 3,
    TFSfourstartotal = TFStotalfours * 4,
    TFSfivestartotal = TFStotalfives * 5,
    ONTtotalfives = sum(ONTFiveStars),
    ONTtotalfours = sum(ONTFourStars),
    ONTtotalthrees = sum(ONTThreeStars),
    ONTthreestartotal = ONTtotalthrees * 3,
    ONTfourstartotal = ONTtotalfours * 4,
    ONTfivestartotal = ONTtotalfives * 5,
  ) |>
  summarise(
    ONTtotalstar = ONTthreestartotal+ONTfourstartotal+ONTfivestartotal,
    TFStotalstar = TFSthreestartotal+TFSfourstartotal+TFSfivestartotal
  )
